reactionary_rebels = {
	icon = 1
	
	area = nation 					# Stays within the nation
	break_alliance_on_win = yes
	
	# <this is the governement in the country> = <this is what the rebels want>
	government = {
		county = county
		feudal_dominion = feudal_monarchy
		lordship = lordship
		duchy = duchy
		daimyo = daimyo
		march = march
		petty_kingdom = petty_kingdom
		beylik = beylik
		principality = principality
		emirate = emirate
		tribal_chiefdom = tribal_chiefdom
		grand_duchy = grand_duchy
		sultanate = sultanate
		feudal_monarchy = feudal_monarchy
        shogunate = shogunate
		elective_monarchy = feudal_monarchy
		khanate = khanate
		empire = empire
		caliphate = caliphate
		peasant_republic = lordship
		military_dictatorship = feudal_monarchy
        democracy = lordship	
        oligarchic_republic = lordship
		confederation = feudal_monarchy
		theocracy = theocracy	
		holy_order = holy_order
	}
	
	
	defection = none					# Will defect to the "best" alternative.
	independence = none				# Does not strive for independence
	defect_delay = 18
	
	ideology = reactionary #royalist
	
	allow_all_cultures = yes
	allow_all_religions = yes
	allow_all_ideologies = yes
	
	resilient = yes					# Disbands on retreat
	reinforcing = yes				# Does not receive reinforcements
	general = yes					# Does use generals
	smart = yes					# Does not assault forts, tactically withdraw from battle, etc.
	unit_transfer = no				# Units do not remain after enforcing demands.
	
	occupation_mult = 1.0			# The amount of strength this rebel gains from taking provinces

	# triggers a general rising, modifier will be multiplied with the countrys army strength and compared to the factions strenght
	will_rise = 
	{
		factor = 0.5
		
		modifier = {
			war = yes
			factor = 0.8
		}
		
		modifier = {
			war_exhaustion = 4
			factor = 0.95
		}
		
		modifier = {
			war_exhaustion = 6
			factor = 0.95
		}
		
		modifier = {
			war_exhaustion = 8
			factor = 0.95
		}
		
		modifier = {
			plurality = 90
			factor = 2
		}
		modifier = {
			plurality = 80
			factor = 1.75
		}
		modifier = {
			plurality = 70
			factor = 1.5
		}
		modifier = {
			plurality = 60
			factor = 1.25
		}
		modifier = {
			NOT = { plurality = 20 }
			factor = 0.95
		}
		modifier = {
			NOT = { plurality = 10 }
			factor = 0.80
		}
		modifier = {
			NOT = { plurality = 5 }
			factor = 0.50
		}
		modifier = {
			NOT = { plurality = 1 }
			factor = 0.1
		}
	}
	
	# The rebel type with the highest modifier for this pop gets picked
	spawn_chance = {
		factor = 1.02
		modifier = {
			factor = 0.01
			NOT = {
				OR = {
					pop_majority_ideology = reactionary
					pop_majority_ideology = fascist
				}
			}
		}
		modifier = {
			factor = 0
			OR = {
				 social_movement = yes
				 political_movement = yes
				}
		}		
		modifier = {
			factor = 1.0115
			pop_majority_ideology = reactionary
		}
		modifier = {
			factor = 1.0115
			NOT = { political_reform_want = 0.05 
					social_reform_want = 0.05 
			}
		}
		modifier = {
			factor = 0
			country = { civilized = no }
		}
		modifier = {
			factor = 0.3
			literacy = 0.2
		}
		modifier = {
			factor = 0.2
			literacy = 0.3
		}
		modifier = {
			factor = 0.3
			AND = {
				pop_majority_ideology = conservative
				NOT = {
					country = {
						has_country_modifier = global_liberal_agitation
						has_country_modifier = springtime_of_nations	
					}
				}
			}
		}
		modifier = {
			factor = 0
			is_primary_culture = no
			is_accepted_culture = no
		}
		modifier = {
			factor = 0.8
			is_accepted_culture = yes
		}
		modifier = {
			factor = 0.2
			country = { government = duchy}
		}
		modifier = {
			factor = 0.2
			country = { government = feudal_monarchy}
		}
		modifier = {
			factor = 1.015
			country = { NOT = { plurality = 35 } }
		}
	}
 
 	#This is checked for EACH province in the Area of Operations
	movement_evaluation = {
		factor = 1
		modifier = {
			factor = 0.1
			units_in_province = 1
		}
		modifier = {
			factor = 0.001
			controlled_by = REB
		}
		modifier = {
			factor = 1.5
			average_militancy = 4
		}
		modifier = {
			factor = 1.5
			average_militancy = 6
		}
		modifier = {
			factor = 1.5
			average_militancy = 8
		}
		modifier = {
			factor = 2.0
			is_capital = yes
		}
	}
	
 	# Province Scope
 	#siege_won_trigger = {
 	#}
	
	#siege_won_effect = {
	#}
	

	# Country scope
	demands_enforced_trigger = {
		capital_scope = {
			controlled_by = REB
			province_control_days = 60
		}
	}
	
	demands_enforced_effect = {
		political_reform = none_voting
		political_reform = party_appointed
		political_reform = censored_press
		political_reform = no_trade_unions
		political_reform = underground_parties
		political_reform = colonial_transportation
		political_reform = no_parliament
		social_reform = low_minimum_wage

		any_country = { limit = { tag = THIS NOT = { plurality = 40 } } 
                         plurality = -100
                         plurality = 40
		}    
		plurality = -20
		any_pop = { limit = { agree_with_ruling_party = yes } militancy = -4 }
	}
}
